39  /**
40   * Convert native "item" to desc. In this case, the native "item"
41   * is assumed to already be formatted as an OAI <record> element,
42   * with the possible exception that multiple metadataFormats may
43   * be present in the <metadata> element. The "crosswalk", merely
44   * involves pulling out the one that is requested.
45   */
46  public class Copydesc extends XSLTCrosswalk {
47      private Transformer transformer = null;
48      
49      /**
50       * The constructor assigns the schemaLocation associated with this crosswalk. Since
51       * the crosswalk is trivial in this case, no properties are utilized.
52       *
53       * @param properties properties that are needed to configure the crosswalk.
54       */
55      public Copydesc(Properties properties)
56          throws OAIInternalServerError {
57   	super(properties, "http://errol.oclc.org/oai:xmlregistry.oclc.org:sts/desc http://errol.oclc.org/oai:xmlregistry.oclc.org:sts/desc.xsd", null);
58          try {
59              String xsltName = properties.getProperty("Copydesc.xsltName");
60              if (xsltName != null) {
61                  StreamSource xslSource = new StreamSource(new FileInputStream(xsltName));
62                  TransformerFactory tFactory = TransformerFactory.newInstance();
63                  this.transformer = tFactory.newTransformer(xslSource);
64              }
65          } catch (Exception e) {
66              e.printStackTrace();
67              throw new OAIInternalServerError(e.getMessage());
68          }
69      }
70      
71      /**
72       * Can this nativeItem be represented in DC format?
73       * @param nativeItem a record in native format
74       * @return true if DC format is possible, false otherwise.
75       */
76      public boolean isAvailableFor(Object nativeItem) {
77          ArrayList list = (ArrayList)nativeItem;
78          return list.contains("desc");
79      }
80  }
