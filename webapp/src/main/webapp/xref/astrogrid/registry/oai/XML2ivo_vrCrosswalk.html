33  /**
34   * Convert native "item" to oai_dc. In this case, the native "item"
35   * is assumed to already be formatted as an OAI <record> element,
36   * with the possible exception that multiple metadataFormats may
37   * be present in the <metadata> element. The "crosswalk", merely
38   * involves pulling out the one that is requested.
39   */
40  public class XML2ivo_vrCrosswalk extends Crosswalk {
41      private static String elementName = "Resource";
42      private static final String prefix = "vr:";
43      
44      private String versionStr = null;
45      
46      private static final String ASTROGRID_SCHEMA_BASE = "http://software.astrogrid.org/schema/";
47      protected static String schemaLocationBase;
48  
49      //private static final String elementStart = "<" + elementName;
50      //private static final String elementEnd = elementName + ">";
51      
52      /**
53       * Static to be used on the initiatian of this class for the config
54       */   
55      static {
56            if(schemaLocationBase == null) {              
57                schemaLocationBase = ContractsFilter.getContextURL() != null ? ContractsFilter.getContextURL() + "/schema/" :
58                                     ASTROGRID_SCHEMA_BASE;
59            }
60      }  
61      
62  
63      /**
64       * The constructor assigns the schemaLocation associated with this crosswalk. Since
65       * the crosswalk is trivial in this case, no properties are utilized.
66       *
67       * @param properties properties that are needed to configure the crosswalk.
68       */
69      public XML2ivo_vrCrosswalk(Properties properties) {
70       super("http://www.ivoa.net/xml/VOResource/v" + properties.getProperty("registry_version") + " http://www.ivoa.net/xml/VOResource/VOResource-v" + properties.getProperty("registry_version") + ".xsd ");
71       this.versionStr = properties.getProperty("registry_version");
72       
73      }
74  
75      /**
76       * Can this nativeItem be represented in DC format?
77       * @param nativeItem a record in native format
78       * @return true if DC format is possible, false otherwise.
79       */
80      public boolean isAvailableFor(Object nativeItem) {
81          String fullItem = (String)nativeItem;
82          if ((fullItem.indexOf(("<" + elementName)) >= 0) || (fullItem.indexOf(("<" + prefix + ":" + elementName)) >= 0) )
83              return true;
84          return false;
85     }
86  
87      /**
88       * Perform the actual crosswalk.
89       *
90       * @param nativeItem the native "item". In this case, it is
91       * already formatted as an OAI <record> element, with the
92       * possible exception that multiple metadataFormats are
93       * present in the <metadata> element.
94       * @return a String containing the XML to be stored within the <metadata> element.
95       * @exception CannotDisseminateFormatException nativeItem doesn't support this format.
96       */
97      public String createMetadata(Object nativeItem)
98     throws CannotDisseminateFormatException {
99  	   String fullItem = (String)nativeItem;
100 	   try {
101 		   org.w3c.dom.Document doc = DomHelper.newDocument("<wrapper xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\">" + fullItem + "</wrapper>");
102 		   org.w3c.dom.NodeList nl = doc.getElementsByTagNameNS("*","Resource");
103 		   if(nl.getLength() > 0) {
104 			   int tempIndex;
105 			    StringBuffer resContent = new StringBuffer(DomHelper.ElementToString((org.w3c.dom.Element)nl.item(0)));
106 				String temp = resContent.substring(0,resContent.indexOf(">"));
107 		        //see if it has a schemaLocation attribute if not then we need to add it.
108 		          if(temp.indexOf("schemaLocation") == -1 && versionStr.equals("1.0")) {
109 		      		String schemaLocations = null;
110 		            schemaLocations =  "http://www.ivoa.net/xml/RegistryInterface/v1.0 "  + schemaLocationBase + 
111 		                               "registry/RegistryInterface/v1.0/RegistryInterface.xsd " +
112 		                               "http://www.ivoa.net/xml/VOResource/v1.0 " + 
113 		                               schemaLocationBase + "vo-resource-types/VOResource/v1.0/VOResource.xsd ";
114 		        	  if((tempIndex = temp.indexOf("type")) != -1) {
115 		              	tempIndex = temp.indexOf("\"",tempIndex);
116 		              	temp = temp.substring((tempIndex+1),temp.indexOf("\"",tempIndex+1));
117 		                if(temp.endsWith("Registry") || temp.endsWith("Authority")) {
118 		                    schemaLocations += " http://www.ivoa.net/xml/VORegistry/v1.0 " + schemaLocationBase + "vo-resource-types/VORegistry/v1.0/VORegistry.xsd";
119 		                } else {
120 		                    schemaLocations += " http://www.ivoa.net/xml/VODataService/v1.0 " + schemaLocationBase + "vo-resource-types/VODataService/v1.0/VODataService.xsd " +
121 		                    "http://www.ivoa.net/xml/VOTable/v1.0 " + schemaLocationBase + "vo-formats/VOTable/v1.0/VOTable.xsd";
122 	                        schemaLocations += " http://www.ivoa.net/xml/ConeSearch/v1.0 " + schemaLocationBase + "vo-resource-types/ConeSearch/v1.0/ConeSearch.xsd";    
123 	                        schemaLocations += " http://www.ivoa.net/xml/SIA/v1.0 " + schemaLocationBase + "vo-resource-types/SIA/v1.0/SIA.xsd";
124 	                        schemaLocations += " http://www.ivoa.net/xml/SSA/v0.4 " + schemaLocationBase + "vo-resource-types/SSA/v0.4/SSA.xsd";
125 	                        schemaLocations += " urn:astrogrid:schema:vo-resource-types:STAP:v1.0 " + schemaLocationBase + "vo-resource-types/STAP/v1.0/STAP.xsd";
126 	                        schemaLocations += " http://www.ivoa.net/xml/CEA/v1.0rc1 " + schemaLocationBase + "vo-resource-types/CEAService/v1.0rc1/CEAService.xsd";
127 		                }
128 		                //add schemaLocation.
129 		                resContent.insert(resContent.indexOf(">")," xsi:schemaLocation=\"" + schemaLocations + "\"");
130 		        	  }//if type
131 		          }//if schemalocations
132 
133 		          //return DomHelper.ElementToString((org.w3c.dom.Element)nl.item(0));
134 		          return resContent.toString();
135 		   } else {
136 			   throw new CannotDisseminateFormatException(getSchemaLocation());
137 		   }
138 	   }catch(Exception e) {
139 		   e.printStackTrace();
140 		   throw new CannotDisseminateFormatException(getSchemaLocation());
141 	   }
142    
143     }
