116      * Convert the requested 'from' parameter to the finest granularity supported
117      * by this repository.
118      * @exception BadArgumentException one or more of the arguments are bad.
119      */
120     public String toFinestFrom(String from) 
121 	throws BadArgumentException {
122 	if (debug) {
123 	    System.out.println("AbstractCatalog.toFinestFrom: from=" + from);
124 	    System.out.println("                            target=" + VALID_GRANULARITIES[supportedGranularityOffset]);
125 	}
126 	if (from.length() > VALID_GRANULARITIES[supportedGranularityOffset].length()) {
127 	    throw new BadArgumentException();
128 	}
129 	if (from.length() != VALID_GRANULARITIES[supportedGranularityOffset].length()) {
130 	    StringBuffer sb = new StringBuffer(from);
131 	    if (sb.charAt(sb.length()-1) == 'Z')
132 		sb.setLength(sb.length()-1);
133 	    
134 	    sb.append(FROM_GRANULARITIES[supportedGranularityOffset].substring(sb.length()));
135 	    from = sb.toString();
136 	}
137 
138 	if (!isValidGranularity(from)) {
139 	    throw new BadArgumentException();
140 	}
141 
142 	return from;
143     }

263      * Does the specified date conform to the supported granularity of this repository?
264      * @param date a UTC date
265      * @return true if date conforms to the supported granularity of this repository,
266      * false otherwise.
267      */
268     private boolean isValidGranularity(String date) {
269 	if (date.length() > VALID_GRANULARITIES[supportedGranularityOffset].length())
270 	    return false;
271 	
272 	if (date.length() < VALID_GRANULARITIES[0].length()
273 	    || !Character.isDigit(date.charAt(0)) // YYYY
274 	    || !Character.isDigit(date.charAt(1))
275 	    || !Character.isDigit(date.charAt(2))
276 	    || !Character.isDigit(date.charAt(3))
277 	    || date.charAt(4) != '-'
278 	    || !Character.isDigit(date.charAt(5)) // MM
279 	    || !Character.isDigit(date.charAt(6))
280 	    || date.charAt(7) != '-'
281 	    || !Character.isDigit(date.charAt(8)) // DD
282 	    || !Character.isDigit(date.charAt(9))) {
283 	    return false;
284 	}
285 	
286 	if (date.length() > VALID_GRANULARITIES[0].length()) {
287 	    if (date.charAt(10) != 'T'
288 		|| date.charAt(date.length()-1) != 'Z'
289 		|| !Character.isDigit(date.charAt(11)) // hh
290 		|| !Character.isDigit(date.charAt(12))
291 		|| date.charAt(13) != ':'
292 		|| !Character.isDigit(date.charAt(14)) // mm
293 		|| !Character.isDigit(date.charAt(15))

296 // 	    }

347     /**
348      * Factory method for creating an AbstractCatalog instance. The properties
349      * object must contain the following entries:
350      * <ul>
351      *   <li><b>AbstractCatalog.className</b> property which points to a class
352      *       that implements the AbstractCatalog interface. Note that this class
353      *       must have a constructor that accepts a properties object as a
354      *       parameter.</li>
355      *   <li><b>Crosswalks.&lt;supported formats&gt;</b> properties which
356      *       satisfy the constructor for the Crosswalks class</li>
357      * </ul>
358      *
359      * @param properties Properties object containing entries necessary to
360      * initialize the class
361      * to be created.
362      * @return on object instantiating the AbstractCatalog interface.
363      * @exception Throwable some sort of problem occurred.
364      */
365     public static AbstractCatalog factory(Properties properties) throws Throwable {
366 	AbstractCatalog oaiCatalog = null;
367 	String oaiCatalogClassName =
368 	    properties.getProperty("AbstractCatalog.oaiCatalogClassName");
369 	String recordFactoryClassName =
370 	    properties.getProperty("AbstractCatalog.recordFactoryClassName");
371         System.out.println("the catalog name found in the factory = " + oaiCatalogClassName);
372         System.out.println("the record name found in the factory = " + recordFactoryClassName);
373         if (oaiCatalogClassName == null) {
374             throw new ClassNotFoundException(
375                   "AbstractCatalog.oaiCatalogClassName is missing from properties file");
376         }
377         if (recordFactoryClassName == null) {
378             throw new ClassNotFoundException(
379                   "AbstractCatalog.recordFactoryClassName is missing from properties file");
380         }
381         Class oaiCatalogClass = Class.forName(oaiCatalogClassName);
382         Constructor oaiCatalogConstructor =
383             oaiCatalogClass.getConstructor(new Class[] {Properties.class});
384         try {
385             oaiCatalog =
386                 (AbstractCatalog)oaiCatalogConstructor.newInstance(new Object[]
387                     {properties});
388             if (debug) {
389                 System.out.println("AbstractCatalog.factory: recordFactoryClassName="
390                                    + recordFactoryClassName);
391             }
392 	    Class recordFactoryClass = Class.forName(recordFactoryClassName);
393 	    Constructor recordFactoryConstructor =
394 		recordFactoryClass.getConstructor(new Class[] {Properties.class});
395             oaiCatalog.recordFactory =
396 		(RecordFactory)recordFactoryConstructor.newInstance(new Object[] {properties});
397             if (debug) {
398                 System.out.println("AbstractCatalog.factory: recordFactory=" + oaiCatalog.recordFactory);
399             }
400 	    String harvestable = properties.getProperty("AbstractCatalog.harvestable");
401 	    if (harvestable != null && harvestable.equals("false")) {
402 		oaiCatalog.harvestable = false;
403 	    }
404 	    String secondsToLive =
405 		properties.getProperty("AbstractCatalog.secondsToLive");
406 	    if (secondsToLive != null) {
407 		oaiCatalog.millisecondsToLive = Integer.parseInt(secondsToLive) * 1000;
408 	    }
409 	    String granularity = properties.getProperty("AbstractCatalog.granularity");
410 	    for (int i = 0; granularity != null && i < VALID_GRANULARITIES.length; ++i) {
411 		if (granularity.equalsIgnoreCase(VALID_GRANULARITIES[i])) {
412 		    oaiCatalog.supportedGranularityOffset = i;
413 		    break;
414 		}
415 	    }
416 	    if (oaiCatalog.supportedGranularityOffset == -1) {
417 		oaiCatalog.supportedGranularityOffset = 0;
418 		System.err.println("AbstractCatalog.factory: Invalid or missing AbstractCatalog.granularity property. Setting value to default: " +
419 				   VALID_GRANULARITIES[oaiCatalog.supportedGranularityOffset]);
420 	    }
421         } catch (InvocationTargetException e) {
422             throw e.getTargetException();
423         }
424 	return oaiCatalog;
425     }

506     /**
507      * Retrieve a list of records that satisfy the specified criteria
508      *
509      * @param from beginning date in the form of YYYY-MM-DD or null if earliest
510      * date is desired
511      * @param until ending date in the form of YYYY-MM-DD or null if latest
512      * date is desired
513      * @param set set name or null if no set is desired
514      * @return a Map object containing an optional "resumptionToken" key/value
515      * pair and a "records" Iterator object. The "records" Iterator contains a
516      * set of Records objects.
517      * @exception BadArgumentException one or more of the arguments are bad.
518      * @exception CannotDisseminateFormatException the requested metadataPrefix isn't supported
519      * @exception NoItemsMatchException no items fit the criteria
520      * @exception NoSetHierarchyException sets aren't defined for this repository
521      * @exception OAIInternalServerError signals an http status code 500 problem
522      */
523     public Map listRecords(String from, String until, String set, String metadataPrefix)
524         throws BadArgumentException, CannotDisseminateFormatException, NoItemsMatchException,
525 	       NoSetHierarchyException, OAIInternalServerError {
526 	if (debug) {
527 	    System.out.println("in AbstractCatalog.listRecords");
528 	}
529 	Map listIdentifiersMap = listIdentifiers(from, until, set, metadataPrefix);
530 	String resumptionToken = (String)listIdentifiersMap.get("resumptionToken");
531 	Iterator identifiers = (Iterator)listIdentifiersMap.get("identifiers");
532 	
533 	Map listRecordsMap = new HashMap();
534 	ArrayList records = new ArrayList();
535 
536 	while (identifiers.hasNext()) {
537 	    String identifier = (String)identifiers.next();
538 	    try {
539 		records.add(getRecord(identifier, metadataPrefix));
540 	    } catch (IdDoesNotExistException e) {
541 		throw new OAIInternalServerError("GetRecord failed to retrieve identifier '"
542 						 + identifier + "'");
543 	    }
544 	}
545 	listRecordsMap.put("records", records.iterator());
546 	if (resumptionToken != null) {
547 	    listRecordsMap.put("resumptionToken", resumptionToken);
548 	}
549 	return listRecordsMap;
550     }

560      * @exception BadResumptionTokenException The resumptionToken is bad.
561      * @exception OAIInternalServerError signals an http status code 500 problem

563     public Map listRecords(String resumptionToken)
564         throws BadResumptionTokenException, OAIInternalServerError {
565 	Map listIdentifiersMap = listIdentifiers(resumptionToken);
566 	resumptionToken = (String)listIdentifiersMap.get("resumptionToken");
567 	Iterator identifiers = (Iterator)listIdentifiersMap.get("identifiers");
568 	String metadataPrefix = (String)listIdentifiersMap.get("metadataPrefix");
569 	
570 	Map listRecordsMap = new HashMap();
571 	ArrayList records = new ArrayList();
572 
573 	while (identifiers.hasNext()) {
574 	    String identifier = (String)identifiers.next();
575 	    try {
576 		records.add(getRecord(identifier, metadataPrefix));
577 	    } catch (IdDoesNotExistException e) {
578 		throw new OAIInternalServerError("GetRecord failed to retrieve identifier '"
579 						 + identifier + "'");

581 		// someone cheated
582 		throw new BadResumptionTokenException();

586 	if (resumptionToken != null) {
587 	    listRecordsMap.put("resumptionToken", resumptionToken);
588 	}
589 	return listRecordsMap;
590     }

596     public Map getResumptionMap(String resumptionToken, int completeListSize, int cursor) {
597 	Map resumptionMap = null;
598 	if (resumptionToken != null) {
599 	    resumptionMap = new HashMap();
600 	    resumptionMap.put("resumptionToken", resumptionToken);
601 	    if (millisecondsToLive > 0) {
602 		Date now = new Date();
603 		Date then = new Date((new Date()).getTime() + millisecondsToLive);
604 		resumptionMap.put("expirationDate", ServerVerb.createResponseDate(then));
605 	    }
606 	    if (completeListSize >= 0) {
