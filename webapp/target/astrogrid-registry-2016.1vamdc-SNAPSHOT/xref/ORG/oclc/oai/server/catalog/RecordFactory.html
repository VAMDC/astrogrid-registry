92      /**
93       * Get a list of supported schemaLocations for the specified native record.
94       * @param nativeItem native database record
95       * @return A Vector containing all the schemaLocations this record can support.
96       * @exception NoMetadatdaFormatsException This record doesn't support any of the
97       * available schemaLocations for this repository.
98       */
99      public Vector getSchemaLocations(Object nativeItem)
100    throws NoMetadataFormatsException {
101    if (isDeleted(nativeItem)) {
102        throw new NoMetadataFormatsException();
103    }
104    Vector v = new Vector();
105    Iterator iterator = getCrosswalks().iterator();
106    while (iterator.hasNext()) {
107        Map.Entry entry = (Map.Entry)iterator.next();
108        CrosswalkItem crosswalkItem = (CrosswalkItem)entry.getValue();
109        Crosswalk crosswalk = crosswalkItem.getCrosswalk();
110        if (crosswalk.isAvailableFor(getMetaDataFromMap(nativeItem))) {
111       v.add(crosswalk.getSchemaLocation());
112        }
113    }
114    return v;
115     }

222      * Create the &lt;record&gt; for the given record and metadataFormat selection.
223      * Use this version of create if the setSpecs are derived from a source other
224      * other than the nativeItem itself.
225      *
226      * @param nativeItem the native record
227      * @param schemaURL the schemaURL desired for the response
228      * @param the metadataPrefix from the request
229      * @return a String containing the OAI record response.
230      * @exception IllegalArgumentException One of the header components for this record is bad.
231      * @exception This nativeItem doesn't support the specified metadataPrefix
232      */
233     public String create(Object nativeItem, String schemaURL, String metadataPrefix,
234           Iterator setSpecs, Iterator abouts)
235    throws IllegalArgumentException, CannotDisseminateFormatException {
236         try {
237             if (isDeleted(nativeItem)) {
238                 StringBuffer sb = new StringBuffer("<oai:record>");
239                 sb.append(createHeader(nativeItem)[0]);
240                 sb.append("</oai:record>");
241                 return sb.toString();
242             }
243             String result = quickCreate(nativeItem, schemaURL, metadataPrefix);
244             if (result == null) {
245                 if (setSpecs == null) {
246                     setSpecs = getSetSpecs(nativeItem);
247                 }
248                 if (abouts == null) {
249                     abouts = getAbouts(nativeItem);
250                 }
251                 result = create(nativeItem, schemaURL,
252                                 getOAIIdentifier(nativeItem),
253                                 getDatestamp(nativeItem),
254                                 setSpecs,
255                                 abouts,
256                                 isDeleted(nativeItem));
257             }
258 
259             return result;
260         } catch (OAIInternalServerError e) {
261             e.printStackTrace();
262             throw new CannotDisseminateFormatException(e.getMessage());
263         }
