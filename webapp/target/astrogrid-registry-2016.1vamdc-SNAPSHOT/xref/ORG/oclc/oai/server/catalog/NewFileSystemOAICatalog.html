117     private static ArrayList getSets(Properties properties) {
118         TreeMap treeMap = new TreeMap();
119 	String propertyPrefix = "Sets.";
120 	Enumeration propNames = properties.propertyNames();
121 	while (propNames.hasMoreElements()) {
122 	    String propertyName = (String)propNames.nextElement();
123 	    if (propertyName.startsWith(propertyPrefix)) {
124                 treeMap.put(propertyName, properties.get(propertyName));
125 	    }
126 	}
127  	return new ArrayList(treeMap.values());
128     }

130     private void loadFileMap(int homeDirLen, File currentDir)
131         throws IOException {
132         try {
133             String[] list = currentDir.list();
134             DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
135             DocumentBuilder builder = factory.newDocumentBuilder();
136             for (int i=0; i<list.length; ++i) {
137                 File child = new File(currentDir, list[i]);
138                 if (child.isDirectory() && !"CVS".equals(child.getName())) {
139                     loadFileMap(homeDirLen, child);
140                 } else if (isMetadataFile(child)) {
141                     String path = file2path(homeDirLen, child);
142                     System.out.println("parsing " + path);
143                     //                 String datestamp = date2OAIDatestamp(new Date(child.lastModified()));
144                     File file = localIdentifier2File(path);
145                     FileInputStream fis = new FileInputStream(file);
146                     InputSource data = new InputSource(fis);
147                     Node doc = builder.parse(data);
148                     fis.close();
149                     
150                     Node datestampNode = XPathAPI.selectSingleNode(doc, "/record/header/datestamp");
151                     fileDateMap.put(path,
152                                     XPathAPI.eval(datestampNode, "string()").str());
153                     NodeList setNodes = XPathAPI.selectNodeList(doc, "/record/header/setSpec");
154                     for (int j=0; j<setNodes.getLength(); ++j) {
155                         Node setSpecNode = setNodes.item(j);
156                         String setSpec = XPathAPI.eval(setSpecNode,
157                                                        "string()").str();
158                         ArrayList setSpecList = (ArrayList)setMap.get(setSpec);
159                         if (setSpecList == null) {
160                             setSpecList = new ArrayList();
161                             setMap.put(setSpec, setSpecList);
162                         }
163                         setSpecList.add(path);
164                     }
165                 }
166             }
167         } catch (Exception e) {
168             e.printStackTrace();
169             throw new IOException(e.getMessage());
170         }
171     }

209     private HashMap getNativeHeader(String path)
210         throws IOException {
211         HashMap recordMap = null;
212         if (fileDateMap.containsKey(path)) {
213             recordMap = new HashMap();
214             recordMap.put("localIdentifier", path.substring(0, path.lastIndexOf(".")));
215             recordMap.put("lastModified", fileDateMap.get(path));
216             ArrayList setSpecs = new ArrayList();
217             Iterator keySet = setMap.keySet().iterator();
218             while (keySet.hasNext()) {
219                 String key = (String)keySet.next();
220                 ArrayList identifierList = (ArrayList)setMap.get(key);
221                 if (identifierList.contains(path)) {
222                     setSpecs.add(key);
223                 }
224             }
225             recordMap.put("setSpecs", setSpecs.iterator());
226             return recordMap;
227         }
228         return recordMap;
229     }

246     private HashMap getNativeRecord(String path)
247         throws IOException {
248         HashMap recordMap = getNativeHeader(path);
249         if (recordMap == null) {
250             return null;
251         } else {
252             File file = localIdentifier2File(path);
253             try {
254                 FileInputStream fis = new FileInputStream(file);
255                 BufferedInputStream bis = new BufferedInputStream(fis);
256                 byte[] buffer = new byte[(int)file.length()];
257                 bis.read(buffer, 0, (int)file.length());
258                 recordMap.put("recordBytes", buffer);
259                 bis.close();
260                 fis.close();
261                 return recordMap;
262             } catch (FileNotFoundException e) {
263                 return null;
264             }
265         }
266     }
